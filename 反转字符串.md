class Solution {
    public String reverseString(String s) {
        if(s.equals(""))
            return s;
        if(s.length()==1)
            return s;
        char[] sc =s.toCharArray();
        int  index = 0;
        int  end   = sc.length-1;
        char temp  = 'c';
        do {
            temp      = sc[index];
            sc[index] = sc[end];
            sc[end]   = temp;
            index++;
            end--;
        } while (index != end && end >= index);
        return new String(sc);
    }
}
